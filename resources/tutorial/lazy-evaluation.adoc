== Lazy Evaluation

[.eval-clojure]
----
(range 0 10)
----

[.eval-clojure]
----
(range 10)
----
[.eval-clojure]
----
#_(range)
----

[.eval-clojure]
----
(take 10 (range))
----

[.eval-clojure]
----
(time (nth (range) 1000000))
----

[.eval-clojure]
----
#_(time (nth (map inc (range)) 1000000))
----

[.eval-clojure]
----
#_(first (filter (fn [number]
          (println number)
          (= number 20))
        (map inc (range))))
----

[.eval-clojure]
----
#_(filter (fn [number]
                 (println number)
                 (= number 20))
               (map inc (range)))
----
